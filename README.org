* Start
Here give I the user the choice of quit the install or continue.
#+begin_src sh 
echo "This will overrides the xinit file (.xinitrc). Take backup."

echo -n  "Will you continue? [Y/n]: "

if [[ $svar == "n" ]] || [[ $svar == "N" ]]
then
    exit
else
